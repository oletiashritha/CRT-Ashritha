package day5;

 // a tread used to perform parllel execution in java  
/*
 * types of thread : both of which use to implement parallel processing in java depending on priority and importance of task
 * 
 * 1.user thread: JVM will wait for all the active user thread to complete their excetion bedore is shutdown itself
 * 
 * 2.demon thread :Doesn't get the performance  
 */

class Demo1 extends Thread{
	public void run()
	{
		for(int i=0;i<=5;i++)
		{
			System.out.println("user thread exceuting");
			try{
				Thread.sleep(1000);
			}
			catch(Exception e){
				System.out.println("some problem will occurred");
			}
		}
	}
}
class Demo2 extends Thread{
	public void run()
	{
		for(int i=0;i<=20;i++)
		{
			System.out.println("Daemon  thread exceuting");
			try{
				Thread.sleep(1000);
			}
			catch(Exception e){
				System.out.println("some problem will occurred");
			}
		}
	}
}
public class Typesofthread {

	public static void main(String[] args) {
		System.out.println("main() started executing");
		Demo1 d1=new Demo1();
		Demo2 d2=new Demo2();
		d2.setDaemon(true);
		d1.start();
		d2.start();
		System.out.println("main() stopped executing");

	}

}
