package AITS;
import java.util.*;

public class ExampleForTreeSet {
	public static void main(String args[]) {
		TreeSet ts=new TreeSet();
		ts.add(10);
		ts.add(20);
		ts.add(30);
		ts.add(40);
		System.out.println("TreeSet: "+ts);
	for(Object i:ts)
	{
		System.out.println(i);
	}
}
}
